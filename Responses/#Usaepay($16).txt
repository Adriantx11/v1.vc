	<script>
	defer(function(){
		$(function(){
			/***************************************************
			 * Tabbed Interface
			 ***************************************************/
			var $tabsets = $('[tabset]');

			$tabsets
				.each(function(){
					var $tabset = $(this);
					var $tabs = $tabset.find('[tab]');
					$tabset
						.on('click', '[tab]', function(){
							var $clicked = $(this);
							activateTab($tabs, $clicked);
						});

					$tabs
						.each(function(){
							var $tab = $(this);

							if(! $tab.hasClass('active')){
								$('[panel="' + $tab.attr('tab') + '"]').hide();
							}

							if($tab.find(':checked').length){
								activateTab($tabs, $tab);
							}
						});
				});

			function activateTab($tabs, $activate){
				$tabs
					.each(function(){
						var $tab = $(this);
						var $panel = $('[panel="' + $tab.attr('tab') + '"]');

						if($tab.is($activate)){
							$tab.addClass('active');
							$panel.show();
						}
						else {
							$tab.removeClass('active');
							$panel.hide();
						}
					});
			}
		});
	});
	</script>
